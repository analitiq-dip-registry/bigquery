# THIS connector's own driver and dialect packages (the engine pins no
# database drivers). The ADBC BigQuery driver exchanges Arrow buffers
# natively; adbc-driver-manager is the shared ADBC runtime.
adbc-driver-bigquery>=1.3.0
adbc-driver-manager>=1.1.0

# sqlalchemy-bigquery registers the "bigquery" SQLAlchemy dialect. The data
# transport is ADBC, but the engine's shared read path
# (GenericSQLConnector.read_batches -> QueryBuilder) borrows SQLAlchemy Core's
# compiler to render paged SELECTs and resolves the dialect via
# sqlalchemy.dialects.registry.load("bigquery"). "bigquery" is not a
# SQLAlchemy built-in, so without this package every source read raises
# ImportError before the first page. Floor is the first release with
# SQLAlchemy 2.0 support (the engine CDK pins sqlalchemy>=2.0,<3).
sqlalchemy-bigquery>=1.8.0
